16       protected String getRealPathFromURI(Context context, Uri contentUri) { 
17           Cursor cursor = null; 
18           try { 
19               String[] proj = { MediaStore.Images.Media.DATA }; 
20               cursor = context.getContentResolver().query(contentUri,  proj, null, null, null); 
21               int column_index = cursor.getColumnIndexOrThrow(MediaStore.Images.Media.DATA); 
22               cursor.moveToFirst(); 
23               return cursor.getString(column_index); 
24           } catch (Exception e) {

27           } finally {

34       public static Bitmap modifyOrientation(Bitmap bitmap, String image_absolute_path) throws IOException { 
35           ExifInterface ei = new ExifInterface(image_absolute_path); 
36           int orientation = ei.getAttributeInt(ExifInterface.TAG_ORIENTATION, ExifInterface.ORIENTATION_NORMAL); 
37    
38           switch (orientation) { 
39               case ExifInterface.ORIENTATION_ROTATE_90: 
40                   return rotate(bitmap, 90); 
41    
42               case ExifInterface.ORIENTATION_ROTATE_180: 
43                   return rotate(bitmap, 180); 
44    
45               case ExifInterface.ORIENTATION_ROTATE_270: 
46                   return rotate(bitmap, 270); 
47    
48               case ExifInterface.ORIENTATION_FLIP_HORIZONTAL: 
49                   return flip(bitmap, true, false); 
50    
51               case ExifInterface.ORIENTATION_FLIP_VERTICAL: 
52                   return flip(bitmap, false, true); 
53    
54               default: 
55                   return bitmap; 
56           } 
57       }

59       public static Bitmap rotate(Bitmap bitmap, float degrees) { 
60           Matrix matrix = new Matrix(); 
61           matrix.postRotate(degrees); 
62           return Bitmap.createBitmap(bitmap, 0, 0, bitmap.getWidth(), bitmap.getHeight(), matrix, true); 
63       } 
64    
65       public static Bitmap flip(Bitmap bitmap, boolean horizontal, boolean vertical) { 
66           Matrix matrix = new Matrix(); 
67           matrix.preScale(horizontal ? -1 : 1, vertical ? -1 : 1); 
68           return Bitmap.createBitmap(bitmap, 0, 0, bitmap.getWidth(), bitmap.getHeight(), matrix, true); 
69       }

71       public Uri getImageUri(Context inContext, Bitmap inImage) { 
72           ByteArrayOutputStream bytes = new ByteArrayOutputStream(); 
73           inImage.compress(Bitmap.CompressFormat.JPEG, 100, bytes); 
74           String path = MediaStore.Images.Media.insertImage(inContext.getContentResolver(), inImage, "Title", null); 
75           return Uri.parse(path); 
76       }
