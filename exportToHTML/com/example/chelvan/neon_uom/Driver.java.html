435      private void animateCamera(@NonNull Location location) { 
436          LatLng latLng = new LatLng(location.getLatitude(), location.getLongitude()); 
437          mMap.animateCamera(CameraUpdateFactory.newCameraPosition(getCameraPositionWithBearing(latLng))); 
438      } 
439   
440      @NonNull 
441      private CameraPosition getCameraPositionWithBearing(LatLng latLng) { 
442          return new CameraPosition.Builder().target(latLng).zoom(16).build(); 
443      } 
444   
445      private void showMarker(@NonNull Location currentLocation) { 
446          LatLng latLng = new LatLng(currentLocation.getLatitude(), currentLocation.getLongitude()); 
447          if (marker == null) { 
448              marker = mMap.addMarker(new MarkerOptions().icon(BitmapDescriptorFactory.fromResource(R.drawable.busicon)).position(latLng)); 
449          } else { 
450              com.example.chelvan.neon_uom.MarkerAnimation.animateMarkerToGB(marker, latLng, new LatLngInterpolator.Spherical()); 
451          } 
452      }
