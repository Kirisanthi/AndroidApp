156      protected boolean isOnline() { 
157          ConnectivityManager cm = (ConnectivityManager)getSystemService(Context.CONNECTIVITY_SERVICE); 
158          NetworkInfo netInfo = cm.getActiveNetworkInfo(); 
159          if (netInfo != null && netInfo.isConnectedOrConnecting()) { 
160              return true; 
161          } else { 
162              return false; 
163          }
