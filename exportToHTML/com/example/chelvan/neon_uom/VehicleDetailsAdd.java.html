262      protected boolean isOnline() { 
263          ConnectivityManager cm = (ConnectivityManager)getSystemService(Context.CONNECTIVITY_SERVICE); 
264          NetworkInfo netInfo = cm.getActiveNetworkInfo(); 
265   
266          if (netInfo != null && netInfo.isConnectedOrConnecting() ) { 
267              return true; 
268          }

271              return false;
