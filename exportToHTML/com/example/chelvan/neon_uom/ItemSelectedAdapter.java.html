43       @Override 
44       public View getView(int position, View view, ViewGroup viewGroup) { 
45           vi = view; 
46           //Populate the Listview 
47           final int pos = position; 
48           Item items = data.get(pos); 
49           if(view == null) { 
50               vi = inflater.inflate(R.layout.listview_item, null); 
51               viewHolder = new ViewHolder(); 
52               viewHolder.checkBox = (CheckBox) vi.findViewById(R.id.checkbox); 
53               viewHolder.name = (TextView) vi.findViewById(R.id.name); 
54               vi.setTag(viewHolder); 
55           }else 
56               viewHolder = (ViewHolder) view.getTag(); 
57           viewHolder.name.setText(items.getName()); 
58           if(items.isCheckbox()){ 
59               viewHolder.checkBox.setChecked(true); 
60           } 
61           else { 
62               viewHolder.checkBox.setChecked(false); 
63           } 
64           return vi;
