149  class UsersAdapter extends ArrayAdapter<FeedbackData> { 
150      public UsersAdapter(Context context, ArrayList<FeedbackData> users) { 
151          super(context, 0, users); 
152      } 
153   
154      @Override 
155      public View getView(int position, View convertView, ViewGroup parent) { 
156          // Get the data item for this position 
157          FeedbackData user = getItem(position); 
158          // Check if an existing view is being reused, otherwise inflate the view 
159          if (convertView == null) { 
160              convertView = LayoutInflater.from(getContext()).inflate(R.layout.feedbackdata, parent, false); 
161          } 
162          // Lookup view for data population 
163          TextView tvName = (TextView) convertView.findViewById(R.id.tvName); 
164          TextView tvHome = (TextView) convertView.findViewById(R.id.tvHome); 
165          // Populate the data into the template view using the data object 
166          tvName.setText(user.points); 
167          tvHome.setText(user.compliments); 
168          // Return the completed view to render on screen 
169          return convertView; 
170      }
