226      private void chooseImage() { 
227          Intent intent = new Intent(); 
228          intent.setType("image/*"); 
229          intent.setAction(Intent.ACTION_GET_CONTENT); 
230          startActivityForResult(Intent.createChooser(intent, "Select Picture"), PICK_IMAGE_REQUEST);

232      }

234      @Override 
235      protected void onActivityResult(int requestCode, int resultCode, Intent data) { 
236          super.onActivityResult(requestCode, resultCode, data); 
237          if(requestCode == PICK_IMAGE_REQUEST && resultCode == RESULT_OK 
238                  && data != null && data.getData() != null ) 
239          { 
240              filePath = data.getData(); 
241              IamgeHandling im = new IamgeHandling(); 
242   
243              try { 
244                  Bitmap bitmap = MediaStore.Images.Media.getBitmap(getContentResolver(),filePath); 
245                  String filepp = im.getRealPathFromURI(UserDetailsAdd.this,filePath); 
246                  Bitmap bitmapmodified = im.modifyOrientation(bitmap,filepp); 
247   
248                  profile.setImageBitmap(bitmapmodified); 
249                  uploadImage(); 
250              } 
251              catch (IOException e) 
252              { 
253                  Toast.makeText(UserDetailsAdd.this,""+e,Toast.LENGTH_SHORT).show(); 
254                  e.printStackTrace(); 
255              } 
256          } 
257      }

260      private void uploadImage() { 
261   
262          if(filePath != null) 
263          { 
264              final ProgressDialog progressDialog = new ProgressDialog(this); 
265              progressDialog.setTitle("Uploading..."); 
266              progressDialog.show(); 
267   
268              final StorageReference ref = storageReference.child("images/"+ email); 
269              ref.putFile(filePath) 
270                      .addOnSuccessListener(new OnSuccessListener<UploadTask.TaskSnapshot>() { 
271                          @Override 
272                          public void onSuccess(UploadTask.TaskSnapshot taskSnapshot) { 
273                              progressDialog.dismiss(); 
274                              profile_url = ref.getRoot().toString(); 
275   
276   
277                          } 
278                      }) 
279                      .addOnFailureListener(new OnFailureListener() { 
280                          @Override 
281                          public void onFailure(@NonNull Exception e) { 
282                              progressDialog.dismiss(); 
283                              Toast.makeText(UserDetailsAdd.this, "Failed "+e.getMessage(), Toast.LENGTH_SHORT).show(); 
284                          } 
285                      }) 
286                      .addOnProgressListener(new OnProgressListener<UploadTask.TaskSnapshot>() { 
287                          @Override 
288                          public void onProgress(UploadTask.TaskSnapshot taskSnapshot) { 
289                              double progress = (100.0*taskSnapshot.getBytesTransferred()/taskSnapshot 
290                                      .getTotalByteCount()); 
291                              progressDialog.setMessage("Uploaded "+(int)progress+"%"); 
292                          } 
293                      }); 
294          } 
295

298      protected boolean isOnline() { 
299          ConnectivityManager cm = (ConnectivityManager)getSystemService(Context.CONNECTIVITY_SERVICE); 
300          NetworkInfo netInfo = cm.getActiveNetworkInfo(); 
301   
302          if (netInfo != null && netInfo.isConnectedOrConnecting() ) { 
303              return true; 
304          } 
305   
306          else { 
307              return false; 
308          }
