13   public class MarkerAnimation {

15       public static void animateMarkerToGB(final Marker marker, final LatLng finalPosition, final LatLngInterpolator.Spherical latLngInterpolator) { 
16           final LatLng startPosition = marker.getPosition(); 
17           final Handler handler = new Handler(); 
18           final long start = SystemClock.uptimeMillis(); 
19           final Interpolator interpolator = new AccelerateDecelerateInterpolator(); 
20           final float durationInMs = 2000; 
21    
22           handler.post(new Runnable() { 
23               long elapsed; 
24               float t; 
25               float v; 
26    
27               @Override 
28               public void run() { 
29                   // Calculate progress using interpolator 
30                   elapsed = SystemClock.uptimeMillis() - start; 
31                   t = elapsed / durationInMs; 
32                   v = interpolator.getInterpolation(t); 
33    
34                   marker.setPosition(latLngInterpolator.interpolate(v, startPosition, finalPosition)); 
35    
36                   // Repeat till progress is complete. 
37                   if (t < 1) { 
38                       // Post again 16ms later. 
39                       handler.postDelayed(this, 16); 
40                   } 
41               } 
42           }); 
43       }
