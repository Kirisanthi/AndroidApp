3    import com.google.android.gms.maps.model.LatLng;

5    import static java.lang.Math.asin; 
6    import static java.lang.Math.atan2; 
7    import static java.lang.Math.cos; 
8    import static java.lang.Math.pow; 
9    import static java.lang.Math.sin; 
10   import static java.lang.Math.sqrt; 
11   import static java.lang.Math.toDegrees; 
12   import static java.lang.Math.toRadians; 
13    
14   public interface LatLngInterpolator { 
15    
16       LatLng interpolate(float fraction, LatLng a, LatLng b); 
17    
18       class Spherical implements LatLngInterpolator { 
19    
20           /* From github.com/googlemaps/android-maps-utils */ 
21           @Override 
22           public LatLng interpolate(float fraction, LatLng from, LatLng to) { 
23               // http://en.wikipedia.org/wiki/Slerp 
24               double fromLat = toRadians(from.latitude); 
25               double fromLng = toRadians(from.longitude); 
26               double toLat = toRadians(to.latitude); 
27               double toLng = toRadians(to.longitude); 
28               double cosFromLat = cos(fromLat); 
29               double cosToLat = cos(toLat); 
30    
31               // Computes Spherical interpolation coefficients. 
32               double angle = computeAngleBetween(fromLat, fromLng, toLat, toLng); 
33               double sinAngle = sin(angle); 
34               if (sinAngle < 1E-6) { 
35                   return from; 
36               } 
37               double a = sin((1 - fraction) * angle) / sinAngle; 
38               double b = sin(fraction * angle) / sinAngle; 
39    
40               // Converts from polar to vector and interpolate. 
41               double x = a * cosFromLat * cos(fromLng) + b * cosToLat * cos(toLng); 
42               double y = a * cosFromLat * sin(fromLng) + b * cosToLat * sin(toLng); 
43               double z = a * sin(fromLat) + b * sin(toLat); 
44    
45               // Converts interpolated vector back to polar. 
46               double lat = atan2(z, sqrt(x * x + y * y)); 
47               double lng = atan2(y, x); 
48               return new LatLng(toDegrees(lat), toDegrees(lng)); 
49           } 
50    
51           private double computeAngleBetween(double fromLat, double fromLng, double toLat, double toLng) { 
52               // Haversine's formula 
53               double dLat = fromLat - toLat; 
54               double dLng = fromLng - toLng; 
55               return 2 * asin(sqrt(pow(sin(dLat / 2), 2) + 
56                       cos(fromLat) * cos(toLat) * pow(sin(dLng / 2), 2))); 
57           } 
58       } 
59   }
